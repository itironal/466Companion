# Binarize a grayscale image 
import matplotlib.pyplot as plt
from skimage import data, color, io
import urllib.request


thresh = 200
image = data.camera()
thresh = 150
# Load image from an external URL and save locally
url = "https://raw.githubusercontent.com/itironal/466Companion/main/images/yeni.png"
urllib.request.urlretrieve(url, "images/yeni.png")
image = io.imread('images/yeni.png')
binary = image > thresh

fig, axes = plt.subplots(ncols=2, figsize=(12, 5))
ax = axes.ravel()

ax[0].imshow(image, cmap=plt.cm.gray);
ax[0].set_title('Original image');
ax[0].axis('off');

ax[1].imshow(binary, cmap=plt.cm.gray);
ax[1].set_title('Thresholded image');
ax[1].axis('off');

plt.show()

## Power-law transform

import matplotlib.pyplot as plt
from skimage import data

c = 1
low_gamma = 0.5
high_gamma = 2

#img = data.moon()
url = "https://raw.githubusercontent.com/itironal/466Companion/main/images/cathedral.jpg"
urllib.request.urlretrieve(url, "images/cathedral.jpg")
img = io.imread('images/cathedral.jpg')
# Convert the image to float and scale it to [0,1]
img = img.astype(float)
img = (img-img.min())/img.max()

img_low_gamma = c*img**low_gamma;
img_high_gamma = c*img**high_gamma;

fig, axes = plt.subplots(ncols=3, figsize=(20, 10))
ax = axes.ravel()

ax[0].imshow(img, cmap='gray');
ax[0].set_title('Original image');
ax[0].axis('off');

ax[1].imshow(img_low_gamma, cmap='gray');
ax[1].set_title('Power law transform (gamma < 1)');
ax[1].axis('off');

ax[2].imshow(img_high_gamma, cmap='gray');
ax[2].set_title('Power law transform (gamma > 1)');
ax[2].axis('off');

Programming exercise 5.6:#

  1. Run the following code to see the output of CLAHE.

import cv2 as cv
import numpy as np
import matplotlib.pyplot as plt
import urllib.request
from PIL import Image

# read original image
url = "https://raw.githubusercontent.com/itironal/466Companion/main/images/dervis_hocam_gray.png"
original_img = np.array(Image.open(urllib.request.urlopen(url)))[:,:,0]

plt.figure(figsize=(15,10))

plt.rcParams.update({'font.size': 18})
plt.rcParams['xtick.labelsize'] = 14
plt.rcParams['ytick.labelsize'] = 14

#show original image
plt.subplot(2,3,1)
plt.imshow(original_img, cmap='gray')
plt.title('Original Image', y = -0.2)
plt.axis('off')

# create original images's histogram
(n, bins) = np.histogram(original_img.ravel(), bins=256, range=(0, 255))
plt.subplot(2,3,4)
plt.bar(bins[:-1], n, width = 1)
plt.title('Original Histogram', y = -0.2)

# calculate cumulative histogram
cumulative_hist = np.cumsum(n)
equalized_img = np.zeros(original_img.shape, dtype=np.uint)

# create equalized image using cumulative histogram
size = float(original_img.shape[0] * original_img.shape[1])
coefficient = 255.0 / size
for x in range(original_img.shape[0]):
    for y in range(original_img.shape[1]):
        equalized_img[x][y] = int(coefficient * cumulative_hist[original_img[x][y]])

# show equalized image
plt.subplot(2,3,2)
plt.imshow(equalized_img, cmap='gray')
plt.title('Equalized Image', y = -0.2)
plt.axis('off')

# create equalized image's histogram
(n, bins) = np.histogram(equalized_img.ravel(), bins=256, range=(0, 255))
plt.subplot(2,3,5)
plt.bar(bins[:-1], n, width = 1)
plt.title('Equalized Histogram', y = -0.2)

# create clahe image
clahe = cv.createCLAHE(clipLimit=2.0, tileGridSize=(8,8))
clahe_img = clahe.apply(original_img)

# show clahe image
plt.subplot(2,3,3)
plt.imshow(clahe_img, cmap='gray')
plt.title('Clahe Image', y = -0.2)
plt.axis('off')

# create clahe image's histogram
(n, bins) = np.histogram(clahe_img.ravel(), bins=256, range=(0, 255))
plt.subplot(2,3,6)
plt.bar(bins[:-1], n, width = 1)
plt.title('Clahe Histogram', y = -0.2)
plt.show()
